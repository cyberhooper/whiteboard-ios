314 /* Called when the user selects a property of a person in their address book (ex. phone, email, location,...)
315  This method will allow them to send a text or email inviting them to Anypic. */
316 - (BOOL)peoplePickerNavigationController:(ABPeoplePickerNavigationController *)peoplePicker shouldContinueAfterSelectingPerson:(ABRecordRef)person property:(ABPropertyID)property identifier:(ABMultiValueIdentifier)identifier {
317 
318  if (property == kABPersonEmailProperty) {
319 
320  ABMultiValueRef emailProperty = ABRecordCopyValue(person,property);
321  NSString *email = (__bridge_transfer NSString *)ABMultiValueCopyValueAtIndex(emailProperty,identifier);
322  self.selectedEmailAddress = email;
323  if ([MFMailComposeViewController canSendMail]) {
324  // go directly to mail
325  [self presentMailComposeViewController:email];
326  } else if ([MFMessageComposeViewController canSendText]) {
327  // go directly to iMessage
328  [self presentMessageComposeViewController:email];
329  }
330 
331  } else if (property == kABPersonPhoneProperty) {
332  ABMultiValueRef phoneProperty = ABRecordCopyValue(person,property);
333  NSString *phone = (__bridge_transfer NSString *)ABMultiValueCopyValueAtIndex(phoneProperty,identifier);
334 
335  if ([MFMessageComposeViewController canSendText]) {
336  [self presentMessageComposeViewController:phone];
337  }
338  }
339 
340  return NO;
341 }

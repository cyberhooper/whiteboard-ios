165 - (void)sectionHeaderLikesButtonPressed:(WBPhotoTimelineSectionHeaderView *)sectionView {
166  WBPhoto *photo = ((WBPhoto *)[self.photos objectAtIndex:sectionView.sectionIndex.intValue]);
167  if (![photo.likes containsObject:[WBDataSource currentUser].userID]) {
168  [self likePhoto:photo completion:^(BOOL success) {
169  if (success) {
170  sectionView.numberOfLikes = @(sectionView.numberOfLikes.intValue + 1);
171  sectionView.isLiked = YES;
172  }
173  }];
174  } else {
175  [self unlikePhoto:photo completion:^(BOOL success) {
176  if (success && sectionView.numberOfLikes > 0) {
177  sectionView.numberOfLikes = @(sectionView.numberOfLikes.intValue - 1);
178  sectionView.isLiked = NO;
179  }
180  }];
181  }
182  NSLog(@"Likes pressed");
183 }
